import asyncio
import aiohttp
import re
import os

# Fontes Globais de Néctar
SOURCES = [
    "https://api.proxyscrape.com/v2/?request=displayproxies&protocol=http&timeout=10000&country=all&ssl=all&anonymity=elite",
    "https://www.proxy-list.download/api/v1/get?type=http&anon=elite",
    "https://raw.githubusercontent.com/TheSpeedX/SOCKS-List/master/http.txt",
    "https://raw.githubusercontent.com/monosans/proxy-list/main/proxies/http.txt",
    "https://raw.githubusercontent.com/fate0/proxylist/master/proxy.list",
    "https://raw.githubusercontent.com/rooster666/proxy-list/main/http.txt"
]

# Países Alvo (ISO Codes)
TARGET_COUNTRIES = ['BR', 'PT', 'GB'] # Brasil, Portugal, Reino Unido
ELITE_FILE = "elite.txt"

async def check_geo_and_save(session, proxy):
    try:
        # 1. Validação de Elite L1
        async with session.get("http://httpbin.org/get", proxy=f"http://{proxy}", timeout=3) as res:
            if res.status == 200:
                data = await res.json()
                if "Via" not in data['headers'] and "X-Forwarded-For" not in data['headers']:
                    
                    # 2. Identificação de Território (Glitch Geo)
                    async with session.get(f"http://ip-api.com/json/{proxy.split(':')[0]}", timeout=2) as geo_res:
                        geo_data = await geo_res.json()
                        country_code = geo_data.get('countryCode')
                        
                        if country_code in TARGET_COUNTRIES:
                            print(f"[+] NÉCTAR {country_code} IDENTIFICADO: {proxy}")
                            with open(ELITE_FILE, "a") as f:
                                f.write(f"{proxy} # {country_code}\n")
    except:
        pass

async def main():
    while True:
        print(f"\n[!] IMPÉRIO MUTANTE: Filtrando Brasil, Portugal e UK...")
        if os.path.exists(ELITE_FILE): open(ELITE_FILE, 'w').close()

        async with aiohttp.ClientSession() as session:
            all_proxies = []
            for url in SOURCES:
                try:
                    async with session.get(url, timeout=10) as res:
                        all_proxies.extend(re.findall(r'\d+\.\d+\.\d+\.\d+:\d+', await res.text()))
                except: continue
            
            tasks = [check_geo_and_save(session, p) for p in set(all_proxies)]
            await asyncio.gather(*tasks)
        
        print("[*] Ciclo finalizado. Dormindo 15 min...")
        await asyncio.sleep(900)

if __name__ == "__main__":
    asyncio.run(main())
