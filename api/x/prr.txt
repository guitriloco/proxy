class CaptchaGuard:
    def __init__(self, api_key):
        self.api_key = api_key
        self.solver_url = "http://2captcha.com/in.php"
        self.result_url = "http://2captcha.com/res.php"

    def solve_recaptcha(self, site_key, page_url):
        print(f"[!] QUEBRADOR: Desafiando algoritmo em {page_url}...")
        
        # 1. Envia o desafio
        payload = {
            'key': self.api_key,
            'method': 'userrecaptcha',
            'googlekey': site_key,
            'pageurl': page_url,
            'json': 1
        }
        
        try:
            res = requests.post(self.solver_url, data=payload).json()
            if res['status'] != 1: return None
            request_id = res['request']

            # 2. Aguarda o Néctar (A Resposta)
            for _ in range(20):
                time.sleep(5)
                check_url = f"{self.result_url}?key={self.api_key}&action=get&id={request_id}&json=1"
                result = requests.get(check_url).json()
                
                if result['status'] == 1:
                    print("[+] ALGORITMO QUEBRADO: Token obtido.")
                    return result['request']
            return None
        except:
            return None

import os
import signal

class KillSwitch:
    def __init__(self, expected_proxy_ip):
        self.expected_ip = expected_proxy_ip
        self.active = True

    def monitorar(self, browser_process_id):
        print(f"[!] KILL-SWITCH: Monitorando integridade da conexão...")
        while self.active:
            try:
                # Consulta o IP atual através do túnel
                current_ip = requests.get("https://api.ipify.org", timeout=5).text
                
                if current_ip != self.expected_ip:
                    print(f"[!!!] ALERTA DE VAZAMENTO: IP real detectado! Encerrando...")
                    os.kill(browser_process_id, signal.SIGTERM) # Mata o navegador
                    self.active = False
            except:
                # Se a conexão cair totalmente, também fecha por segurança
                os.kill(browser_process_id, signal.SIGTERM)
                self.active = False
