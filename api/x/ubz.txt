import asyncio
import aiohttp
import re
import os

# Fontes de Néctar - Escala Total
SOURCES = [
    "https://api.proxyscrape.com/v2/?request=displayproxies&protocol=http&timeout=10000&country=all&ssl=all&anonymity=elite",
    "https://www.proxy-list.download/api/v1/get?type=http&anon=elite",
    "https://www.proxyscan.io/download?type=http",
    "https://raw.githubusercontent.com/TheSpeedX/SOCKS-List/master/http.txt",
    "https://raw.githubusercontent.com/monosans/proxy-list/main/proxies/http.txt",
    "https://raw.githubusercontent.com/fate0/proxylist/master/proxy.list",
    "https://raw.githubusercontent.com/clarketm/proxy-list/master/proxy-list-raw.txt",
    "https://raw.githubusercontent.com/sunny9577/proxy-scraper/master/proxies.txt",
    "https://raw.githubusercontent.com/ShiftyTR/Proxy-List/master/http.txt",
    "https://raw.githubusercontent.com/rooster666/proxy-list/main/http.txt"
]

TEST_URL = "http://httpbin.org/get"
ELITE_FILE = "elite.txt"

async def testar_e_salvar(session, proxy):
    try:
        async with session.get(TEST_URL, proxy=f"http://{proxy}", timeout=3) as response:
            if response.status == 200:
                data = await response.json()
                headers = data.get("headers", {})
                # Critério Elite L1
                if "Via" not in headers and "X-Forwarded-For" not in headers:
                    print(f"[+] NÉCTAR ELITE: {proxy}")
                    with open(ELITE_FILE, "a") as f:
                        f.write(f"{proxy}\n")
    except:
        pass

async def ciclo_mineracao():
    while True:
        print("\n" + "="*40)
        print("[!] IMPÉRIO MUTANTE: CICLO DE MINERAÇÃO")
        print("="*40)
        
        if os.path.exists(ELITE_FILE):
            open(ELITE_FILE, 'w').close() 

        async with aiohttp.ClientSession() as session:
            all_proxies = []
            for url in SOURCES:
                try:
                    async with session.get(url, timeout=10) as res:
                        text = await res.text()
                        found = re.findall(r'\d+\.\d+\.\d+\.\d+:\d+', text)
                        all_proxies.extend(found)
                except: continue
            
            unique_proxies = list(set(all_proxies))
            print(f"[*] {len(unique_proxies)} IPs brutos. Auditando...")
            
            tasks = [testar_e_salvar(session, proxy) for proxy in unique_proxies]
            await asyncio.gather(*tasks)
            
        print("[*] Ciclo finalizado. Dormindo 15 min...")
        await asyncio.sleep(900)

if __name__ == "__main__":
    try:
        asyncio.run(ciclo_mineracao())
    except KeyboardInterrupt:
        print("[!] Operação encerrada pelo Imperador.")
2. O Check-list de Soberania
Com os arquivos que você subiu, sua estrutura de pastas deve ser:

pro.py: O motor de mineração (use a versão acima).

aa.py: O roteador central (sua ponte para o navegador).

elite.txt: O arquivo que será criado e alimentado automaticamente.

ip.py e gemini...py: Guarde-os como ferramentas de auditoria manual.

3. Execução em Campo
Abra um terminal no seu i7 e rode: python pro.py.

Abra um terminal no seu i5 (ou no mesmo PC) e rode: python aa.py.

Configure o navegador para 127.0.0.1:8080.
